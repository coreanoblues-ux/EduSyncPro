// src/routes/students.ts (혹은 서버 라우트 파일)
import { z } from "zod";

app.get("/students/quick-search", async (req, res) => {
  try {
    const schema = z.object({
      q: z.string().trim().default(""),
      limit: z.coerce.number().int().min(1).max(50).default(20),
      cursor: z.string().optional()
    });
    const { q, limit, cursor } = schema.parse(req.query);

    // 전처리
    const qLike = q.replace(/\s+/g, " ").toLowerCase(); // 이름/학교
    const qDigits = q.replace(/\D/g, "");               // 전화 검색용 (숫자만)

    // where 조건: q 비어있으면 전체에서 페이지네이션
    const where: any = q
      ? {
          OR: [
            { name: { contains: qLike, mode: "insensitive" } },
            { schoolName: { contains: qLike, mode: "insensitive" } },
            (isFinite(+q) ? { gradeLevel: Number(q) } : undefined),
            (qDigits ? { phoneDigits: { contains: qDigits } } : undefined)
          ].filter(Boolean)
        }
      : {};

    const results = await prisma.student.findMany({
      where,
      take: limit,
      ...(cursor ? { skip: 1, cursor: { id: cursor } } : {}),
      orderBy: { id: "asc" },
      select: {
        id: true, name: true, schoolName: true, gradeLevel: true,
        phoneMasked: true  // 없다면 phone만 선택하거나 제외
      }
    });

    const nextCursor = results.length === limit ? results[results.length - 1].id : null;
    res.json({ items: results, nextCursor });
  } catch (err) {
    // 절대 크래시하지 않도록 방어
    console.error("[quick-search] error:", err);
    res.status(200).json({ items: [], nextCursor: null }); // 실패해도 빈 결과 반환
  }
});
